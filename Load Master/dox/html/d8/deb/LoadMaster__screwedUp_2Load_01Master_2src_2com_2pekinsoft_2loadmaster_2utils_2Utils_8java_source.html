159  public static Point getCenterPoint(Dimension container, Dimension window) {
160  int x = container.width / 2;
161  int y = container.height / 2;
162  x = x - window.width / 2;
163  y = y - window.height / 2;
164  x = x < 0 ? 0 : x;
165  y = y < 0 ? 0 : y;
166  return new Point(x, y);
