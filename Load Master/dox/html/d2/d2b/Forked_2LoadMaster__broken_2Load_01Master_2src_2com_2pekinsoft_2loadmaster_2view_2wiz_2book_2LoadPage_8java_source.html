267  protected String validateContents(Component comp, Object o) {
268  // Set the flags to our settings.
269  if ( orderField.getText().length() == 0 )
270  return "Order Number is required.";
271  else if ( tripField.getText().length() == 0 )
272  return "Trip Number is required.";
273  else if ( grossPayField.getText().length() == 0 )
274  return "Gross Truck Pay is required.";
275  else if ( dispMilesField.getText().length() == 0 )
276  return "Dispatched Miles is required.";
277 
278  // If we get to this point, then all required data has been collected
279  //+ and we need to store all data on this form to the settings map.
280  putWizardData("order", orderField.getText());
281  putWizardData("trip", tripField.getText());
282  putWizardData("truck.pay", grossPayField.getText());
283  putWizardData("miles", dispMilesField.getValue());
284  putWizardData("commodity", commdityField.getText());
285  putWizardData("hazmat", hazMatField.isSelected());
286  putWizardData("tarped", tarpedField.isSelected());
287  putWizardData("team", teamField.isSelected());
288  putWizardData("twic", twicField.isSelected());
289  putWizardData("top.customer", topField.isSelected());
290  putWizardData("cbd", cbdField.isSelected());
291  putWizardData("ramps", rampsField.isSelected());
292 
293  // After we have stored all of the collected data for later use, we can
294  //+ return control to the wizard controller class.
295  return null;
