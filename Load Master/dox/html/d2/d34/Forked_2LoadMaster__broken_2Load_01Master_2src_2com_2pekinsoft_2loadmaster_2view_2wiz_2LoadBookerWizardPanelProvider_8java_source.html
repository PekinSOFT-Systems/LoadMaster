224  @Override
225  protected JComponent createPanel(WizardController wc, String string, Map map) {
226  this.settings = map;
227  this.controller = wc;
228 
229  if (string.equalsIgnoreCase("load")) {
230  // Cannot finish the wizard from this screen.
231  controller.setForwardNavigationMode(
232  WizardController.MODE_CAN_CONTINUE);
233 
234  // We are on the load page.
235  return new LoadPage();
236  } else if (string.equalsIgnoreCase("broker")) {
237  // Cannot finish the wizard from this screen.
238  controller.setForwardNavigationMode(
239  WizardController.MODE_CAN_CONTINUE);
240 
241  // We are on the broker page.
242  return new BrokerPage();
243  } else if (string.equalsIgnoreCase("stops")) {
244  // Cannot finish from this page.
245  controller.setForwardNavigationMode(
246  WizardController.MODE_CAN_CONTINUE);
247 
248  // We are on the first stop page, which is the first of two pages
249  //+ for adding stops in the wizard.
250  return new StopsPage();
251  } else if (string.equals("summary")) {
252  // Can finish from this page.
253  controller.setForwardNavigationMode(
254  WizardController.MODE_CAN_FINISH);
255 
256  // We are on the summary page, which is the final page of the wizard,
257  //+ there is nowhere else to go from here.
258  return new SummaryPage(map);
259  } else {
260  // An unknown page ID has been discovered. Typo?
261  throw new Error("Unknown ID " + string + "\nPlease check the "
262  + "spelling\nand try again.");
263  }
264  }

266  @Override
267  protected Object finish(Map settings) {
268  // Once the user clicks finish, we need to create the load record in the
269  //+ loads table, as well as the stop record(s) in the stops table.
270  load = new LoadModel();
271  load.setMiles(Integer.valueOf(settings.get("miles").toString()));
272  load.setCbd(Boolean.parseBoolean(settings.get("cbd").toString()));
273  load.setCommodity(settings.get("commodity").toString());
274  load.setHazMat(Boolean.parseBoolean(settings.get("hazmat").toString()));
275  load.setLtl(Boolean.parseBoolean(settings.get("ltl").toString()));
276  load.setOrder(settings.get("order").toString());
277  load.setRamps(Boolean.parseBoolean(settings.get("ramps").toString()));
278  try {
279  NumberFormat nf = new java.text.DecimalFormat("#,##0.00");
280  Number pay = nf.parse(settings.get("truck.pay").toString());
281  load.setRate(pay.doubleValue());
282  } catch (ParseException ex) {
283  entry.setMessage("Couldn't parse truck.pay value.");
284  entry.setThrown(ex);
285  Starter.logger.error(entry);
286  }
287  load.setTarped(Boolean.parseBoolean(settings.get("tarped").toString()));
288  load.setTeam(Boolean.parseBoolean(settings.get("team").toString()));
289  load.setTrip(settings.get("trip").toString());
290  load.setTwic(Boolean.parseBoolean(settings.get("twic").toString()));
291  load.setTopCust(Boolean.parseBoolean(settings.get(
292  "top.customer").toString()));
293 
294  load.setDispatch(new Date());
295 
296  String broker = settings.get("brokerList").toString();
297  broker = broker.substring(broker.indexOf("(") + 1);
298  broker = broker.substring(0, broker.length() - 1);
299  load.setBroker(Long.valueOf(broker));
300 
301  int stopCount = Starter.props.getPropertyAsInt("stop.count", "0");
302  int x = 0;
303  do {
304  stop = new StopModel();
305  stop = (StopModel)settings.get("stop" + ++x);
306  stops.addNew(stop);
307  load.addStop(stop);
308  } while ( x < stopCount );
309 
310  try {
311  stops.close();
312  } catch ( DataStoreException ex ) {
313  entry.setMessage("An error occurred accessing the stops database.");
314  entry.setThrown(ex);
315  Starter.logger.error(entry);
316 
317  MessageBox.showError(ex, "Stops Database Error");
318  }
319 
320  // The last thing to do is to check to see if the driver is on an active
321  //+ trip.
322  if ( Starter.props.getProperty("load.current", "No Active Load")
323  .equalsIgnoreCase("No Active Load") ) {
324  // If not, see if the driver wants to start this trip.
325  int response = MessageBox.askQuestion("Would you like to start this "
326  + "trip?", "Start Trip", false);
327 
328  if ( response == MessageBox.YES_OPTION ) {
329  StartTripDialog dlg = new StartTripDialog(null, true);
330  dlg.pack();
331  dlg.setOrderNumber(load.getOrder());
332  dlg.setTripNumber(load.getTrip());
333  dlg.setVisible(true);
334 
335  if ( !dlg.isCancelled() ) {
336  load.setStartOdo(dlg.getOdometer());
337 
338  Starter.props.setProperty("load.current", load.getTrip());
339  Starter.props.setPropertyAsInt("stop.count",
340  load.getStopCount() * 2);
341  Starter.props.setPropertyAsInt("load.stop", 0);
342  Starter.props.flush();
343 
344  ReceivablesCtl ctl;
345 
346  try {
347  ctl = new ReceivablesCtl();
348 // ReceivablesTransaction ar = ;
349  ctl.addNew(new ReceivablesTransaction(
350  load.getDispatch(),
351  load.getTrip(),
352  load.getOrder(),
353  load.getRate()));
354  ctl.close();
355  } catch ( DataStoreException ex ) {
356  entry.setMessage("An error occurred accessing the stops "
357  + "database.");
358  entry.setThrown(ex);
359  Starter.logger.error(entry);
360 
361  MessageBox.showError(ex, "Receivables Journal Error");
362  }
363  }
364  }
365  }
366 
367  loads.addNew(load);
368 
369  try {
370  loads.close();
371  } catch ( DataStoreException ex ) {
372  entry.setMessage("An error occurred accessing the loads database.");
373  entry.setThrown(ex);
374  Starter.logger.error(entry);
375 
376  MessageBox.showError(ex, "Stops Database Error");
377  }
378 
379  // Now, we need to see if the user is currently on
380 
381  // We are not returning anything from this method, so just return a null
382  //+ object, as everything that needed to be done was done here.
383  return null;
384  }
