223

231  gl = new EntryCtl();

320  private boolean isDataValid() {
321  double desired = 0.0;
322  double amt = 0.0;
323  if ( advAmountField.getText() != null
324  && !advAmountField.getText().isBlank()
325  && ! advAmountField.getText().isEmpty()) {
326  desired = Double.parseDouble(advAmountField.getText());
327  }
328 
329  if ( receivable != null ) {
