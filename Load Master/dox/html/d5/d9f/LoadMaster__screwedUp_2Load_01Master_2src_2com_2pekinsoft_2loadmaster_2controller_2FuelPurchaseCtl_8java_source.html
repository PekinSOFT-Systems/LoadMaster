137  @Override
138  public boolean postTransactions() throws DataStoreException {
139  // For this, we are going to need to create an EntryCtl object, as well
140  //+ as an EntryModel object.
141  EntryCtl gl = new EntryCtl();
142  EntryModel tx = null;
143 
144  // We also need a return value to send back to the calling method.
145  boolean success = true; // Default to a successful posting procedure.
146 
147  // We need to loop through all of our journal entries, so we can enter
148  //+ our diesel purchase into the General Ledger.
149  for ( int x = 0; x < records.size(); x++ ) {
150  // Check to see if the current record has not yet been posted.
151  if ( !((FuelPurchaseModel)records.get(x)).isPosted() ) {
152  // Since the record has not yet been posted to the GL, we need
153  //+ to do so now.
154  tx = new EntryModel();
155  tx.setAmount(((FuelPurchaseModel)records.get(x)).getGallonsOfDiesel()
156  * ((FuelPurchaseModel)records.get(x)).getPricePerGallonDiesel());
157  tx.setBalanced(false);
158  tx.setCode("DieselPurch");
159  try {
160  tx.setDate(((FuelPurchaseModel)records.get(x)).getDateAsString());
161  } catch ( ParseException ex ) {
162  // If we get a ParseException setting the date, we will log
163  //+ it...
164  entry.setSourceClassName(getClass().getCanonicalName());
165  entry.setSourceMethodName("postTransactions");
166  entry.setParameters(null);
167  entry.setThrown(ex);
168  entry.setMessage(ex.getMessage());
169  entry.setInstant(Instant.now());
170  Starter.logger.error(entry);
171 
172  // ...but we will still set a date for the transaction, we
173  //+ will just make it the current date.
174  tx.setDate(new Date());
175  }
176  tx.setDeductible(true);
177 
178  String desc = "Purchased " + ((FuelPurchaseModel)records.get(x))
179  .getGallonsOfDieselAsString();
180  desc += " gallons at ";
181  desc += ((FuelPurchaseModel)records.get(x)).getLocation();
182 
183  if ( ((FuelPurchaseModel)records.get(x)).getTripNumber()
184  .equalsIgnoreCase("No Active Load") ) {
185  desc += " while not under an active load.";
186  } else {
187  tx.setDescription(" while on Trip # "
188  + ((FuelPurchaseModel)records.get(x)).getTripNumber());
189  }
190  tx.setDescription(desc);
191 
192  tx.setFromAccount(FuelCardModel.ACCOUNT_NUMBER);
193  tx.setToAccount(FuelPurchaseModel.ACCOUNT_NUMBER);
194 
195  // Now that we have created the GL transaction entry, we can add
196  //+ it to the GL.
197  gl.addNew(tx);
198 
199  // Now we have to do it all again for the DEF that may have been
200  //+ purchased at this same fuel stop.
201  if ( ((FuelPurchaseModel)records.get(x)).isDefPurchased() ) {
202  tx.setAmount(((FuelPurchaseModel)records.get(x)).getGallonsOfDef()
203  * ((FuelPurchaseModel)records.get(x)).getPricePerGallonDef());
204  tx.setBalanced(false);
205  tx.setCode("DEFPurch");
206  tx.setDate(((FuelPurchaseModel)records.get(x)).getDate());
207  tx.setDeductible(true);
208 
209  desc = "Purchased " + ((FuelPurchaseModel)records.get(x))
210  .getGallonsOfDefAsString();
211  desc += " gallons at ";
212  desc += ((FuelPurchaseModel)records.get(x)).getLocation();
213 
214  if ( ((FuelPurchaseModel)records.get(x)).getTripNumber()
215  .equalsIgnoreCase("No Active Load") ) {
216  desc += " while not under an active load.";
217  } else {
218  tx.setDescription(" while on Trip # "
219  + ((FuelPurchaseModel)records.get(x)).getTripNumber());
220  }
221  tx.setDescription(desc);
222 
223  tx.setFromAccount(FuelCardModel.ACCOUNT_NUMBER);
224  tx.setToAccount(FuelPurchaseModel.ACCOUNT_NUMBER);
225 
226  // Now that we have created the GL transaction entry, we can add
227  //+ it to the GL.
228  gl.addNew(tx);
229  }
230  } // No `else`, because transactions only need to be posted once.
231  } // Continue until all records have been read and posted.
232 
233  // Finally, we can post the transactions to the General Ledger.
234  gl.close();
