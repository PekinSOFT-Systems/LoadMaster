152  private void doSave() {
153  // In order to get the selected broker, we need to loop through the
154  //+ records to find which record has the selected ID number.
155  if ( !customerList.getSelectedItem().toString().equalsIgnoreCase(
156  "select customer...") ) {
157  String selectedBroker = customerList.getSelectedItem().toString();
158  long brokerID = Long.valueOf(selectedBroker.substring(
159  selectedBroker.indexOf("[") + 1, // Start after (
160  selectedBroker.indexOf("]"))); // End before )
161 
162  try {
163  records.first();
164 
165  for ( int x = 0; x < records.getRecordCount(); x++ ) {
166  CustomerModel c = records.get();
167 
168  if ( brokerID == c.getId() ) {
169  customer = c;
170  break;
171  } else {
172  if ( records.hasNext() )
173  records.next();
174  }
175  }
176 
177  setVisible(false);
178  } catch ( DataStoreException ex ) {
179  lr.setMessage("Something went wrong moving to the next record.");
180  lr.setThrown(ex);
181  Starter.logger.error(lr);
182 
183  // MessageBox.showError(ex, "Database Access");
184 
185  dispose();
186  }
187  }

190  private boolean isDataValid() {
191  lr.setSourceMethodName("isDateValid");
192  lr.setMessage("Validating the input data.");
193  Starter.logger.enter(lr);
194 
195  // Before testing any of the dates, we need to make sure that the user
196  //+ has selected a customer from the list.
197  if ( customerList.getSelectedItem().toString().equals("Select "
198  + "customer...") ) {
199  // No customer has been chosen, so no sense in moving forward with
200  //+ the data validation.
201  lr.setMessage("Validation completed. Returning findings.");
202  Starter.logger.exit(lr, new Object[] {false});
203  return false;
204  }
205 
206  if ( customerList.getSelectedItem().toString().equals("Add a new "
207  + "customer...") ) {
208  NewCustomerDlg dlg = new NewCustomerDlg( null, true );
209  dlg.pack();
210  dlg.setVisible(true);
211 
212  if ( dlg.isCancelled() == false ) {
213  customer = dlg.getCustomer();
214 
215  records.addNew(customer);
216 
217  customerList.addItem(customer.getCompany() + ": "
218  + customer.getCity() + ", " + customer.getState()
219  + " [" + customer.getId() + "]");
220 
221  customerList.setSelectedIndex(customerList.getItemCount() - 1);
222  }
223  }

225  SimpleDateFormat sdf = new SimpleDateFormat("MM/dd/yyyy");
226 
227  // For the stop to be valid, the late date *MUST* be greater than or
228  //+ equal to the early date.
229  if ( lateDate.getDate() != null && earlyDate.getDate() != null ) {
230  if ( lateDate.getDate().compareTo(earlyDate.getDate()) < 0 ) {
231  // Less than zero (0) means that lateDate is BEFORE earlyDate.
232  lr.setMessage("Validation completed. Returning findings.");
233  Starter.logger.exit(lr, new Object[] {false});
234  return false;
235  }
236  }
237 
238  // Create a couple of Date object to hold the early and late times for
239  //+ comparison, and reset the SimpleDateFormat object to only hold the
240  //+ times, in 24-hour format.
241  sdf = new SimpleDateFormat("HH:mm");
242  Date early = null;
243  Date late = null;
244 
245  if ( earlyTime.getText() != null && lateTime.getText() != null ) {
246  String hours = earlyTime.getText().split(":")[0].trim();
247  String minutes = earlyTime.getText().split(":")[1].trim();
248  if ((hours.length() != 2) & (minutes.length() != 2)) return false;
249  hours = lateTime.getText().split(":")[0].trim();
250  minutes = lateTime.getText().split(":")[1].trim();
251  if ((hours.length() != 2) & (minutes.length() != 2)) return false;
252  try {
253  early = sdf.parse(earlyTime.getText());
254  late = sdf.parse(lateTime.getText());
255  } catch ( ParseException ex ) {
256  lr.setMessage("A stop time was not valid.");
257  lr.setThrown(ex);
258  Starter.logger.error(lr);
259  lr.setMessage("Validation completed. Returning findings.");
260  lr.setThrown(null);
261  Starter.logger.exit(lr, new Object[] {false});
262  // We will return false from here so that the data cannot be
263  //+ saved.
264  return false;
265  }
266  }
267 
268  // Make sure that the early and late time objects have been created.
269  if ( early != null && late != null ) {
270  // Now that we know that both times have been successfully created,
271  //+ we need to validate the times, but only if the early and late
272  //+ dates are the same day.
273  if ( lateDate.getDate().compareTo(earlyDate.getDate()) == 0 ) {
274  // Only perform the next text if the early and late dates are on
275  //+ the same date.
276  if ( late.compareTo(early) < 0 ) {
277  // The late time is before the early time.
278  lr.setMessage("Validation completed. Returning findings.");
279  Starter.logger.exit(lr, new Object[] {false});
280  return false;
281  }
282  }
283  }
284 
285  lr.setMessage("Validation completed. Returning findings.");
286  Starter.logger.exit(lr, new Object[] {true});
287  return true;
