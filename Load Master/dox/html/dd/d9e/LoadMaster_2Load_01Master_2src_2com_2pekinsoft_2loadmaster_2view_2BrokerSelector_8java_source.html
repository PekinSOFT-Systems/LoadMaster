103  private void loadBrokerList() {
104  brokerList.removeAllItems();
105  brokerList.addItem("Select Broker/Agent...");
106 
107  ArrayList<BrokerModel> filtered = new ArrayList<>();
108 
109  if ( allFilterOption.isSelected() ) {
110  try {
111  records.first();
112  } catch ( DataStoreException ex ) {
113  lr.setMessage("Something went wrong moving to the next record.");
114  lr.setThrown(ex);
115  Starter.logger.error(lr);
116 
117  MessageBox.showError(ex, "Database Access");
118  }
119 
120  for ( int x = 0; x < records.getRecordCount(); x++ ) {
121  BrokerModel b = records.get();
122 
123  brokerList.addItem(b.getContact() + " (" + b.getId() + ")");
124 
125  try {
126  if (records.hasNext() )
127  records.next();
128  } catch ( DataStoreException ex ) {
129  lr.setMessage("Something went wrong moving to the next record.");
130  lr.setThrown(ex);
131  Starter.logger.error(lr);
132 
133  // MessageBox.showError(ex, "Database Access");
134  }
135  }
136  } else if ( cityFilterOption.isSelected() ) {
137  try {
138  filtered = records.getCompaniesByCity(criteriaField.getText(),
139  LoadMaster.fileProgress);
140  } catch ( DataStoreException ex ) {
141  lr.setMessage("Something went wrong moving to the next record.");
142  lr.setThrown(ex);
143  Starter.logger.error(lr);
144  }
145 
146  if ( filtered != null && filtered.size() > 0 ) {
147  for ( int x = 0; x < filtered.size(); x++ ) {
148  brokerList.addItem(filtered.get(x).getContact() + " ("
149  + filtered.get(x).getId() + ")");
150  }
151  } else
152  MessageBox.showInfo("No matching records found!", "No Records");
153  } else if ( stateFilterOption.isSelected() ) {
154  try {
155  filtered = records.getCompaniesByState(criteriaField.getText(),
156  LoadMaster.fileProgress);
157  } catch ( DataStoreException ex ) {
158  lr.setMessage("Something went wrong moving to the next record.");
159  lr.setThrown(ex);
160  Starter.logger.error(lr);
161  }
162 
163  if ( filtered != null && filtered.size() > 0 ) {
164  for ( int x = 0; x < filtered.size(); x++ ) {
165  brokerList.addItem(filtered.get(x).getContact() + " ("
166  + filtered.get(x).getId() + ")");
167  }
168  } else
169  MessageBox.showInfo("No matching records found!", "No Records");
170  }
171 
172  filtering = false;
173  }

327  private void selectBrokerActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_selectBrokerActionPerformed
328  // In order to get the selected broker, we need to loop through the
329  //+ records to find which record has the selected ID number.
330  if ( !brokerList.getSelectedItem().toString().equalsIgnoreCase(
331  "select broker/agent...") ) {
332  String selectedBroker = brokerList.getSelectedItem().toString();
333  long brokerID = Long.valueOf(selectedBroker.substring(
334  selectedBroker.indexOf("(") + 1, // Start after (
335  selectedBroker.indexOf(")"))); // End before )
336 
337  try {
338  records.first();
339 
340  for ( int x = 0; x < records.getRecordCount(); x++ ) {
341  BrokerModel b = records.get();
342 
343  if ( brokerID == b.getId() ) {
344  broker = b;
345  break;
346  } else {
347  if ( records.hasNext() )
348  records.next();
349  }
350  }
351 
352  setVisible(false);
353  } catch ( DataStoreException ex ) {
354  lr.setMessage("Something went wrong moving to the next record.");
355  lr.setThrown(ex);
356  Starter.logger.error(lr);
357 
358  // MessageBox.showError(ex, "Database Access");
359 
360  dispose();
361  }
362  }
363  }//GEN-LAST:event_selectBrokerActionPerformed
