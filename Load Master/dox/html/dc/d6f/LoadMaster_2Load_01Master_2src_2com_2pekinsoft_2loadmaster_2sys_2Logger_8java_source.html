223  public Logger(String path, int level, boolean append) {
224  // The first thing we are going to do is to set our log field to null.
225  this.log = null; //+ In this way, it is "initialized" even though it
226  //+ initialized to nothing (null), it will prevent some design-time
227  //+ errors from showing up.
228 
229  // Set our logging level to the level provided.
230  this.level = level;
231 
232  // Set our appending ability to the appending ability provided.
233  this.append = append;
234 
235  // In order to ACTUALLY initialize our log field, we will need to
236  //+ enclose it in a try...catch() block. We will also need to do this
237  //+ anytime we use the log field.
238  try {
239  File file = new File(path);
240  if (!file.exists()) {
241  File dirs = new File(path.substring(0, path.indexOf(file.getName())));
242  dirs.mkdirs();
243  }
244  this.log = new FileWriter(path);
245  } catch (IOException ex) {
246  // We are going to simply show a message box to the user explaining
247  //+ that logging setup failed and then we will turn off logging.
248  String ttl = "Logging Setup Failure";
249 
250  MessageBox.showError(ex, ttl);
251 
252  // Now that we've told the calling application that there was an
253  //+ error, we can turn logging off.
254  this.level = OFF;
255  }
256  }
