201  private void selectBrokerActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_selectBrokerActionPerformed
202  // In order to get the selected broker, we need to loop through the
203  //+ records to find which record has the selected ID number.
204  if ( !brokerList.getSelectedItem().toString().equalsIgnoreCase(
205  "select broker/agent...") ) {
206  String selectedBroker = brokerList.getSelectedItem().toString();
207  long brokerID = Long.valueOf(selectedBroker.substring(
208  selectedBroker.indexOf("(") + 1, // Start after (
209  selectedBroker.indexOf(")"))); // End before )
210 
211  try {
212  records.first();
213 
214  for ( int x = 0; x < records.getRecordCount(); x++ ) {
215  BrokerModel b = records.get();
216 
217  if ( brokerID == b.getId() ) {
218  broker = b;
219  break;
220  } else {
221  if ( records.hasNext() )
222  records.next();
223  }
224  }
225 
226  setVisible(false);
227  } catch ( DataStoreException ex ) {
228  lr.setMessage("Something went wrong moving to the next record.");
229  lr.setThrown(ex);
230  Starter.logger.error(lr);
231 
232  // MessageBox.showError(ex, "Database Access");
233 
234  dispose();
235  }
236  }
237  }//GEN-LAST:event_selectBrokerActionPerformed
