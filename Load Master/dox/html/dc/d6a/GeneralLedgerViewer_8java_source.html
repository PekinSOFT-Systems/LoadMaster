58  try {
59  EntryCtl ctl = new EntryCtl();
60 
61  for ( int x = 0; x < ctl.getRecordCount(); x++ ) {
62  EntryModel data = ctl.get(x);
63 
64  ChartCtl coa = new ChartCtl();
65  String fromAcct = "";
66  String toAcct = "";
67 
68  // Get the name of the account for the from and the to accounts.
69  for ( int y = 0; y < coa.getRecordCount(); y++ ) {
70  if ( coa.get(y).getNumber() == ctl.get(x).getFromAccount() )
71  fromAcct = coa.get(y).getName();
72  if ( coa.get(y).getNumber() == ctl.get(x).getToAccount() )
73  toAcct = coa.get(y).getName();
74  }
75 
76  Object[] row = {data.getDateAsString(), // Tx date
77  data.getCode(), // Tx code
78  data.getDescription(), // Description of Tx
79  fromAcct, // From account
80  toAcct, // To account
81  data.getAmount(), // Tx amount
82  data.isDeductible(), // Tax deductible
83  data.isBalanced() // Balanced with bank.
84  };
85  model.addRow(row);
86  }
87  } catch ( DataStoreException ex ) {
88  MessageBox.showWarning("Something happened while reading in the General Ledger", "Error");
