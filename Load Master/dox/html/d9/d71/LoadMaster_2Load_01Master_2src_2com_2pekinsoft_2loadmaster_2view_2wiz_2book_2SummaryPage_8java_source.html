171  // To list out the stops, we want to list only the stop number, the
172  //+ company, street, suite, city, state, Zip Code, and phone number. In
173  //+ order to accomplish this, we will need to loop through all keys in
174  //+ the data map, appending the stop number to the word "stop". Once we
175  //+ have the customer ID for the stop, we will need to get that customer
176  //+ record from the database. To accomplish this, we will need to get
177  //+ the data from the customers table using a CustomerCtl object and a
178  //+ long value of the String of the customer ID stored in the data map.
179  //+ We will use a CustomerModel object to hold the record that matches
180  //+ the customer we are seeking.
181 
182  try {
183  table = new CustomerCtl();
184 
185  // If we are successful in opening the data table, we can then
186  //+ search for the customer we need. However, we need to loop through
187  //+ all of the stops the user entered to get the customer ID numbers.
188  for ( int x = 0; x < Starter.props.getPropertyAsInt("stop.count",
189  "0"); x++) {
190  int stopNum = x + 1;
191  Object row = map.get("stop" + stopNum);
192  StopModel stop = (StopModel)row;
193  long desiredID = stop.getCustomer();
194 
195  // Now that we have the customer ID we need to match, we need to
196  //+ loop through all of the records in the customer table until
197  //+ we find that customer record.
198  do {
199  if ( table.get().getId() == desiredID ) {
200  cust = table.get();
201 
202  // Exit the loop.
203  break;
204  } else
205  table.next();
206  } while ( table.hasNext() );
207 
208  // Now that we have gotten the appropriate customer information
209  //+ for the stop, we can add the info to the summary page.
210  summary.append("<tr><td>");
211  summary.append(stop.getStopNumber()).append("</td><td>");
212  summary.append(cust.getCompany()).append("</td><td>");
213 
214  SimpleDateFormat sdf = new SimpleDateFormat("MM/dd/yyyy");
215 
216  summary.append(sdf.format(stop.getEarlyDate())).append("</td><td>");
217  summary.append(stop.getEarlyTime()).append("</td><td>");
218  summary.append(sdf.format(stop.getLateDate())).append("</td><td>");
219  summary.append(stop.getLateTime()).append("</td></tr>");
220 // summary.append(cust.getZip()).append("</td><td>");
221 // summary.append(cust.getPhone()).append("</td></tr>");
222  }
223  } catch ( DataStoreException ex ) {
224  entry.setThrown(ex);
225  Starter.logger.error(entry);
226  }
227

229  //+ close out our table and our document.
