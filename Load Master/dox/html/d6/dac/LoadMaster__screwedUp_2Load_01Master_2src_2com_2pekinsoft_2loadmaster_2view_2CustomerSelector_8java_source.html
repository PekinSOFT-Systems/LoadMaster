213  private void doSave() {
214  // In order to get the selected broker, we need to loop through the
215  //+ records to find which record has the selected ID number.
216  if ( !customerList.getSelectedItem().toString().equalsIgnoreCase(
217  "select customer...") ) {
218  String selectedBroker = customerList.getSelectedItem().toString();
219  long brokerID = Long.valueOf(selectedBroker.substring(
220  selectedBroker.indexOf("[") + 1, // Start after (
221  selectedBroker.indexOf("]"))); // End before )
222 
223  try {
224  records.first();
225 
226  for ( int x = 0; x < records.getRecordCount(); x++ ) {
227  CustomerModel c = records.get();
228 
229  if ( brokerID == c.getId() ) {
230  customer = c;
231  break;
232  } else {
233  if ( records.hasNext() )
234  records.next();
235  }
236  }
237

239  } catch ( DataStoreException ex ) {

251  private boolean isDataValid() {
252  lr.setSourceMethodName("isDateValid");
253  lr.setMessage("Validating the input data.");
254  Starter.logger.enter(lr);
255 
256  // Before testing any of the dates, we need to make sure that the user
257  //+ has selected a customer from the list.
258  if ( customerList.getSelectedItem().toString().equals("Select "
259  + "customer...") ) {
260  // No customer has been chosen, so no sense in moving forward with
261  //+ the data validation.
262  lr.setMessage("Validation completed. Returning findings.");
263  Starter.logger.exit(lr, new Object[] {false});
264  return false;
265  }
266 
267  if ( customerList.getSelectedItem().toString().equals("Add a new "
268  + "customer...") ) {
269  NewCustomerDlg dlg = new NewCustomerDlg( null, true );
270  dlg.pack();
271  dlg.setVisible(true);
272 
273  if ( dlg.isCancelled() == false ) {
274  customer = dlg.getCustomer();
275 
276  records.addNew(customer);
277 
278  customerList.addItem(customer.getCompany() + ": "
279  + customer.getCity() + ", " + customer.getState()
280  + " [" + customer.getId() + "]");
281 
282  customerList.setSelectedIndex(customerList.getItemCount() - 1);
283  }
284  }

286  SimpleDateFormat sdf = new SimpleDateFormat("MM/dd/yyyy");
287 
288  // For the stop to be valid, the late date *MUST* be greater than or
289  //+ equal to the early date.
290  if ( lateDate.getDate() != null && earlyDate.getDate() != null ) {
291  if ( lateDate.getDate().compareTo(earlyDate.getDate()) < 0 ) {
292  // Less than zero (0) means that lateDate is BEFORE earlyDate.
293  lr.setMessage("Validation completed. Returning findings.");
294  Starter.logger.exit(lr, new Object[] {false});
295  return false;
296  }
297  }
298 
299  // Create a couple of Date object to hold the early and late times for
300  //+ comparison, and reset the SimpleDateFormat object to only hold the
301  //+ times, in 24-hour format.
302  sdf = new SimpleDateFormat("HH:mm");
303  Date early = null;
304  Date late = null;
305 
306  if ( earlyTime.getText() != null && lateTime.getText() != null ) {
307  String hours = earlyTime.getText().split(":")[0].trim();
308  String minutes = earlyTime.getText().split(":")[1].trim();
309  if ((hours.length() != 2) & (minutes.length() != 2)) return false;
310  hours = lateTime.getText().split(":")[0].trim();
311  minutes = lateTime.getText().split(":")[1].trim();
312  if ((hours.length() != 2) & (minutes.length() != 2)) return false;
313  try {
314  early = sdf.parse(earlyTime.getText());
315  late = sdf.parse(lateTime.getText());
316  } catch ( ParseException ex ) {
317  lr.setMessage("A stop time was not valid.");
318  lr.setThrown(ex);
319  Starter.logger.error(lr);
320  lr.setMessage("Validation completed. Returning findings.");
321  lr.setThrown(null);
322  Starter.logger.exit(lr, new Object[] {false});
323  // We will return false from here so that the data cannot be
324  //+ saved.
325  return false;
326  }
327  }
328 
329  // Make sure that the early and late time objects have been created.
330  if ( early != null && late != null ) {
331  // Now that we know that both times have been successfully created,
332  //+ we need to validate the times, but only if the early and late
333  //+ dates are the same day.
334  if ( lateDate.getDate().compareTo(earlyDate.getDate()) == 0 ) {
335  // Only perform the next text if the early and late dates are on
336  //+ the same date.
337  if ( late.compareTo(early) < 0 ) {
338  // The late time is before the early time.
339  lr.setMessage("Validation completed. Returning findings.");
340  Starter.logger.exit(lr, new Object[] {false});
341  return false;
342  }
343  }
344  }
345 
346  lr.setMessage("Validation completed. Returning findings.");
347  Starter.logger.exit(lr, new Object[] {true});
348  return true;
349  }
