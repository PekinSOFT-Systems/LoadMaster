135  @Override
136  public boolean postTransactions() throws DataStoreException {
137  // For this, we are going to need to create an EntryCtl object, as well
138  //+ as an GLEntryTransaction object.
139  EntryCtl gl = new EntryCtl();
140  GLEntryTransaction tx = null;
141 
142  // We also need a return value to send back to the calling method.
143  boolean success = true; // Default to a successful posting procedure.
144 
145  // We need to loop through all of our journal entries, so we can enter
146  //+ our diesel purchase into the General Ledger.
147  for ( int x = 0; x < records.size(); x++ ) {
148  FuelPurchaseTransaction exp = (FuelPurchaseTransaction)records.get(x);
149 
150  // Check to see if the current record has not yet been posted.
151  if ( !exp.isPosted() ) {
152  // Since the record has not yet been posted to the GL, we need
153  //+ to do so now.
154  tx = new GLEntryTransaction();
155  tx.setAmount(exp.getGallonsOfDiesel()
156  * exp.getPricePerGallonDiesel());
157  tx.setBalanced(false);
158  tx.setCode("DieselPurch");
159  try {
160  tx.setDate(exp.getDateAsString());
161  } catch ( ParseException ex ) {
162  // If we get a ParseException setting the date, we will log
163  //+ it...
164  entry.setSourceClassName(getClass().getCanonicalName());
165  entry.setSourceMethodName("postTransactions");
166  entry.setParameters(null);
167  entry.setThrown(ex);
168  entry.setMessage(ex.getMessage());
169  entry.setInstant(Instant.now());
170  Starter.logger.error(entry);
171 
172  // ...but we will still set a date for the transaction, we
173  //+ will just make it the current date.
174  tx.setDate(new Date());
175  }
176  tx.setDeductible(true);
177 
178  String desc = "Purchased " + exp
179  .getGallonsOfDieselAsString();
180  desc += " gallons at ";
181  desc += exp.getLocation();
182 
183  if ( exp.getTripNumber()
184  .equalsIgnoreCase("No Active Load") ) {
185  desc += " while not under an active load.";
186  } else {
187  tx.setDescription(" while on Trip # "
188  + exp.getTripNumber());
189  }
190  tx.setDescription(desc);
191 
192  tx.setFromAccount(exp.getFromAccount());
193  tx.setToAccount(exp.getAccountNumber());
194 
195  // Now that we have created the GL transaction entry, we can add
196  //+ it to the GL.
197  gl.addNew(tx);
198 
199  // Now we have to do it all again for the DEF that may have been
200  //+ purchased at this same fuel stop.
201  if ( exp.isDefPurchased() ) {
202  tx.setAmount(exp.getGallonsOfDef()
203  * exp.getPricePerGallonDef());
204  tx.setBalanced(false);
205  tx.setCode("DEFPurch");
206  tx.setDate(exp.getDate());
207  tx.setDeductible(true);
208 
209  desc = "Purchased " + exp
210  .getGallonsOfDefAsString();
211  desc += " gallons at ";
212  desc += exp.getLocation();
213 
214  if ( exp.getTripNumber()
215  .equalsIgnoreCase("No Active Load") ) {
216  desc += " while not under an active load.";
217  } else {
218  tx.setDescription(" while on Trip # "
219  + exp.getTripNumber());
220  }
221  tx.setDescription(desc);
222 
223  tx.setFromAccount(exp.getFromAccount());
224  tx.setToAccount(exp.getAccountNumber());
225 
226  // Now that we have created the GL transaction entry, we can add
227  //+ it to the GL.
228  gl.addNew(tx);
229  }
230  } // No `else`, because transactions only need to be posted once.
231  } // Continue until all records have been read and posted.
232 
233  // Finally, we can post the transactions to the General Ledger.
234  gl.close();
235

237  return success;
