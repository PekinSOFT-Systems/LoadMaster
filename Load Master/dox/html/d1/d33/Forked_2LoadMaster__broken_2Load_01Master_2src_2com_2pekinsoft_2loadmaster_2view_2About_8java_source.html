268  private void websiteLabelMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_websiteLabelMouseClicked
269  String[] commands = null;
270  String os = System.getProperty("os.name");
271  if (os.contains("Unix")) commands = new String[] {"firefox ", "konqueror ", "mozilla ", "opera "};
272  if (os.contains("Linux")) commands = new String[] {"firefox ", "google-chrome ", "mozilla ", "opera "};
273  if (os.contains("Windows")) commands = new String[] {"cmd.exe /c start "};
274  if (os.contains("Macintosh")) commands = new String[] {"open "};
275  if (commands == null) return;
276  for (String command : commands) {
277  String systemCommand = command + Starter.props.getProjectWebsite();
278  try {
279  Process process = Runtime.getRuntime().exec(systemCommand);
280  process.waitFor();
281  int exit = process.exitValue();
282  if (exit == 0) return;
283  } catch(Exception e) { e.printStackTrace(); }
284  }
285 
286  }//GEN-LAST:event_websiteLabelMouseClicked
