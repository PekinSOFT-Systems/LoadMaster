100  private void doSave() {
101 
102 
103  if ( isDataValid() ) {
104  model = new ExpenseTransaction(accountNumber);
105 
106  model.setAmount(Double.parseDouble(amountField.getText()));
107  model.setDate(dateField.getDate());
108  model.setDescription(descriptionField.getText());
109  model.setFromAccount(getAccountNumber());
110  model.setId(Long.parseLong(idText.getText()));
111  model.setPosted(false);
112 
113  doClose();
114  } else {
115  switch ( onCtrl++ ) {
116  case 0:
117  fromAccountList.requestFocusInWindow();
118  break;
119  case 1:
120  personalCheckBox.requestFocusInWindow();
121  break;
122  case 2:
123  dateField.requestFocusInWindow();
124  break;
125  case 3:
126  amountField.requestFocusInWindow();
127  break;
128  case 4:
129  descriptionField.requestFocusInWindow();
130  onCtrl = 0;
131  break;
132  }
133  }
134  }

136  private boolean isDataValid() {
137  return (amountField.getText() != null
138  && !amountField.getText().isBlank()
139  || !amountField.getText().isEmpty())
140  && dateField.getDate() != null
141  && !fromAccountList.getSelectedItem().toString()
142  .equalsIgnoreCase("Select pay account...")
143  && (descriptionField.getText() != null
144  && !descriptionField.getText().isBlank()
145  || !descriptionField.getText().isEmpty());
146  }

310  private void ctrlFocused(java.awt.event.FocusEvent evt) {//GEN-FIRST:event_ctrlFocused
311  if ( evt.getSource() instanceof javax.swing.JComboBox )
312  onCtrl = 0;
313  else if ( evt.getSource() instanceof javax.swing.JCheckBox )
314  onCtrl = 1;
