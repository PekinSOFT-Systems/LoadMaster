616  private void connect() throws DataStoreException {
617  BufferedReader in;
618  if ( LoadMaster.fileProgress != null ) {
619  LoadMaster.fileProgress.setMinimum(0);
620  LoadMaster.fileProgress.setMaximum(
621  Starter.props.getPropertyAsInt("table.customers.records", "0"));

624 
625  try {

627 
628  String line = in.readLine();
629

631  String[] record = line.split("~");
632 
633  createAndAddRecord(record);

635  line = in.readLine();

638  LoadMaster.fileProgress.setValue(

645  in.close();
